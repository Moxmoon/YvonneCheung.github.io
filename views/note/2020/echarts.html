Echarts 基础图形自定义绘图

Spt 27, 2020
前端
echarts

Echarts样式、格式自定义配置，以下为部分代码

# 雷达图

# 1. 雷达图配置

 const option = {
      radar: {
          splitNumber: 3, // 雷达的图圈数设置
          name: {
            textStyle: {
              fontSize: '13',
              fontWeight: 500,
              borderRadius: 3,
              padding: [3, 6]
            }
          },
          splitArea: {
            show: false  // 不显示雷达图背景的间隔
          },
          indicator: data //数据
        },
 }

# 2. 雷达图内容配置

 const option = {
     series: [{
          name: '技能',
          type: 'radar',
          // 拐点设置
          symbol: 'circle',
          symbolSize: 8,
          itemStyle: {
            normal: {
              color: '#f3a4a4'
            }
          },
          lineStyle: {
            width: 2
          },
          // 内容区域设置
          areaStyle: {
            color: '#fbe6e6' // 区域背景颜色
          },
          data: [{value: data}] // 数据
        }]
 }

# 柱形图

# 1. 柱形图坐标轴配置

const option = {
     xAxis: {
          name: '科目',
          type: 'category',
          data: xData,
          axisLabel: {
            interval: 0 // 横轴信息全部显示
          },
          axisLine: {
            show: false // 隐藏x轴线
          },
          axisTick: {
            show: false // 隐藏x轴刻度线
          }
        },
     yAxis: {
          name: '得分',
          type: 'value',
          show: true,
          max: 100, // 设置最大值
          min: 0,
          axisLabel: {
            show: true,
            interval: 'auto',
          },
          axisLine: {
            show: false // 隐藏y轴线
          },
          axisTick: {
            show: false // 隐藏y轴刻度线
          },
          // 设置网格线类型 dotted：虚线 solid:实线
          splitLine: { 
            show: true, 
            lineStyle: {
              type: 'dashed'
            }
          }
        }
}

# 2. 柱形图数据配置

自定义柱形图上方文字

const option = {
    series: [{
          type: 'bar',
          barWidth: 30,
          color: '#a7dcc4',
         // 使用markPoint绘制顶部文字
          markPoint: {
            // 形状：'circle','rect', 'roundRect', 'triangle', 'diamond', 'pin', 'arrow'
            // 也可使用图片'image://url'或者矢量路径'path://'
            symbol: 'circle', 
            symbolSize: [45, 25],
            symbolOffset: [20, -25],
            itemStyle: {
              color: '#ff8a90' // 背景颜色
            },
            data: weakData // 可以计算，格式{ value: '薄弱', xAxis: index, yAxis: score }
          }
        }
}

# 折线图

# 1. 鼠标悬浮提示框

有tooltip:{}即可显示，若内容不符合需求，可加formatter自定义，加上html模板，也可以自定义样式和方法

tooltip: {
     formatter: (params) => {
         	const name = `<p><label>考试名称：</label><span class="pc-text-rtl">${this.examData[params.dataIndex]}</span></p>`
            const data = `<p><label>得分：</label><span>${params.data}</span></p>`
            const contact = `${name} ${data}`
            const html = `<div class="chart-popover">${contact}</div>`
            return html
        },
}

# 2. 自定义拐点颜色和对应的图例显示

const option = {
     color: ['#ff8a90', '#a7dcc4'], // 一定要与图例的color对应
     legend: {
          color: ['#ff8a90', '#a7dcc4'],
          data: ['最低分', '最高分'], // 与series的name和个数对应
          right: '9%',
          icon: 'circle',
          itemWidth: 6,
          selectedMode: false // 取消图例上的点击事件
        },
    series: [{
          name: '最低分',
          data: scoreData,
          type: 'line',
          symbol: 'circle',
          symbolSize: 8,
          smooth: 0.6, // 折线平滑度
          lineStyle: {
            color: '#9fc9f5',// 折线颜色
            width: 3
          },
          itemStyle: {
            normal: {
              // color: '#9fc9f5', // 常规拐点颜色
              // 自定义设置最高分和最低分拐点颜色
              color: function (param) {
                if (param.data == maxValue) {
                  return '#a7dcc4'
                } else if (param.data == minValue) {
                  return '#ff8a90'
                } else {
                  return '#9fc9f5'
                }
              }
            }
          },
          areaStyle: { color: '#dbecfd' } //区域颜色
        }, {
          name: '最高分',
          type: 'line'
        }]
}
